CREATE TABLE EMPLEADO
(
    NOMBRE varchar(50),
    ID SERIAL PRIMARY KEY,
    FECHA_INGRESO varchar(50),
    SALARIO bigint
);

CREATE TABLE SOLICITUD (
   CODIGO varchar(50),
   DESCRIPCION varchar(50),
   RESUMEN varchar(50),
   ID SERIAL PRIMARY KEY,
   ID_EMPLEADO bigint,
   FOREIGN KEY(ID_EMPLEADO) REFERENCES EMPLEADO(ID)
);
